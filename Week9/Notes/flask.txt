With "get":
    request.args
    retrieves data
    used in urls
    search results
    for "world" as default use:
        (python) name = request.args.get("name", "world")

With "post":
    request.form
    submit data
    handles larger data
    secure data
    Use:
        (python) return render_template("greet.html", name=request.form.get("name"))
        (html) hello, {% if name %}{{ name }}{% else %}world{% endif %}
